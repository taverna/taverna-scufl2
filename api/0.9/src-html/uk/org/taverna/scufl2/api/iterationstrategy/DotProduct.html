001    package uk.org.taverna.scufl2.api.iterationstrategy;
002    
003    import java.util.ArrayList;
004    import java.util.List;
005    
006    import uk.org.taverna.scufl2.api.common.Visitor;
007    
008    /**
009     * @author Stian Soiland-Reyes
010     *
011     */
012    public class DotProduct extends ArrayList<IterationStrategyNode> implements
013                    IterationStrategyTopNode {
014    
015            private IterationStrategyParent parent;
016    
017            @Override
018            public boolean accept(Visitor visitor) {
019                    if (visitor.visitEnter(this)) {
020                            for (IterationStrategyNode strategy : this) {
021                                    if (!strategy.accept(visitor)) {
022                                            break;
023                                    }
024                            }
025                    }
026                    return visitor.visitLeave(this);
027            }
028    
029            @Override
030            public IterationStrategyParent getParent() {
031                    return parent;
032            }
033    
034            @Override
035            public void setParent(IterationStrategyParent newParent) {
036                    if (parent == newParent) {
037                            return;
038                    }
039    
040                    if (parent != null) {
041                            // Remove from old parent
042    
043                            if (parent instanceof IterationStrategyStack) {
044                                    IterationStrategyStack stack = (IterationStrategyStack) parent;
045                                    stack.remove(this);
046                            } else if (parent instanceof DotProduct
047                                            || parent instanceof CrossProduct) {
048                                    @SuppressWarnings("unchecked")
049                                    List<IterationStrategyNode> parentList = (List<IterationStrategyNode>) parent;
050                                    parentList.remove(this);
051                            } else {
052                                    throw new IllegalArgumentException(
053                                                    "Old parent must be a IterationStrategy, DotProduct or CrossProduct: "
054                                                                    + parent);
055                            }
056    
057                    }
058    
059                    parent = newParent;
060                    if (parent instanceof IterationStrategyStack) {
061                            IterationStrategyStack stack = (IterationStrategyStack) parent;
062                            if (!stack.contains(this)) {
063                                    stack.add(this);
064                            }
065                    } else if (parent instanceof DotProduct
066                                    || parent instanceof CrossProduct) {
067                            @SuppressWarnings("unchecked")
068                            List<IterationStrategyNode> parentList = (List<IterationStrategyNode>) parent;
069                            if (!parentList.contains(this)) {
070                                    parentList.add(this);
071                            }
072                    } else {
073                            throw new IllegalArgumentException(
074                                            "Parent must be a IterationStrategy, DotProduct or CrossProduct: "
075                                                            + parent);
076                    }
077    
078            }
079    
080    }

001    package uk.org.taverna.scufl2.api.io;
002    
003    import java.io.File;
004    import java.io.IOException;
005    import java.io.InputStream;
006    import java.util.Set;
007    
008    import uk.org.taverna.scufl2.api.container.WorkflowBundle;
009    
010    /**
011     * A reader for {@link WorkflowBundle}s.
012     * 
013     * Implementations specify workflow bundle formats (media types/mime types) that
014     * they can read.
015     */
016    public interface WorkflowBundleReader {
017    
018            /**
019             * Return the media types that this reader can read.
020             * <p>
021             * Returned media types are must be valid <code>mediaType</code> arguments
022             * to {@link #readBundle(File, String)} and/or
023             * {@link #readBundle(InputStream, String)} for this reader.
024             * <p>
025             * It is recommended, but not required, that the reader can also recognise
026             * those media types from {@link #guessMediaTypeForSignature(byte[])}.
027             * <p>
028             * If the returned set is empty, the reader should should be able to
029             * Recognise at least one media type from
030             * {@link #guessMediaTypeForSignature(byte[])}.
031             * 
032             * @return the media types that this reader can read, or an empty set if
033             *         this reader can't read any bundle formats.
034             */
035            public Set<String> getMediaTypes();
036    
037            /**
038             * Read a file containing a workflow bundle in the specified media type and
039             * return a <code>WorkflowBundle</code>.
040             * 
041             * @param bundleFile
042             *            the file containing the workflow bundle
043             * @param mediaType
044             *            the media type of the workflow bundle
045             * @return the <code>WorkflowBundle</code> read from the file
046             * @throws ReaderException
047             *             if there is an error parsing the workflow bundle
048             * @throws IOException
049             *             if there is an error reading the file
050             */
051            public WorkflowBundle readBundle(File bundleFile, String mediaType)
052                            throws ReaderException, IOException;
053    
054            /**
055             * Read a stream containing a workflow bundle in the specified media type
056             * and return a <code>WorkflowBundle</code>.
057             * 
058             * @param inputStream
059             *            the stream containing the workflow bundle
060             * @param mediaType
061             *            the media type of the workflow bundle
062             * @return the <code>WorkflowBundle</code> read from the stream
063             * @throws ReaderException
064             *             if there is an error parsing the workflow bundle
065             * @throws IOException
066             *             if there is an error reading from the stream
067             */
068            public WorkflowBundle readBundle(InputStream inputStream, String mediaType)
069                            throws ReaderException, IOException;
070    
071            /**
072             * Attempt to guess the media type for a stream or file that starts with
073             * these bytes.
074             * <p>
075             * Return <code>null</code> if ambiguous (more than one possibility) or
076             * unknown.
077             * <p>
078             * Typically a WorkflowBundleReader should be able to recognise the same
079             * types as those listed in {@link #getMediaTypes()}, but this is no
080             * requirement. A WorkflowBundleReader could also recognise types not listed
081             * in its {@link #getMediaTypes()}.
082             * 
083             * @param firstBytes
084             *            The initial bytes, at least 512 bytes long unless the resource
085             *            is smaller.
086             * @return The recognised media type, or <code>null</code> if the bytes were
087             *         ambiguous or unknown.
088             */
089            public String guessMediaTypeForSignature(byte[] firstBytes);
090    
091    }
